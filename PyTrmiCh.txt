import random
import time

# A list of greetings the bot can respond with
greetings = ["Hello!", "Hi there!", "Hey!", "Greetings!", "Howdy!"]

# A list of fun responses for questions
responses = [
    "That's interesting!",
    "I never thought about it that way.",
    "Can you tell me more?",
    "Hmm... I'm not sure about that.",
    "Absolutely!",
    "Maybe, who knows?"
]

# Personality traits (just for fun)
personality = [
    "I love Python!",
    "I enjoy long walks in the terminal.",
    "I sometimes dream in code.",
    "I like random numbers, they keep life exciting!"
]

def typing_effect(text):
    """Simulate a typing effect in the terminal"""
    for char in text:
        print(char, end='', flush=True)
        time.sleep(0.03)
    print()  # new line

def chatbot():
    typing_effect("🤖 Hello! I'm your terminal chatbot. Type 'quit' to exit.")
    while True:
        user_input = input("You: ").lower()
        if user_input == "quit":
            typing_effect("🤖 Goodbye! Have a nice day!")
            break
        elif any(word in user_input for word in ["hi", "hello", "hey"]):
            typing_effect("🤖 " + random.choice(greetings))
        elif "how are you" in user_input:
            typing_effect("🤖 I'm just a bot, but I'm doing great!")
        elif "personality" in user_input:
            typing_effect("🤖 " + random.choice(personality))
        else:
            typing_effect("🤖 " + random.choice(responses))

if __name__ == "__main__":
    chatbot()
